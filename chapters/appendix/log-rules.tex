\section*{Arithmetic Rules}
\begin{align}
\label{eq:log-sum}
\ln(x\cdot y)&=\ln(x)+\ln(y),\\
\label{eq:log-power}
\ln(x^r)&=r\cdot\ln(x).
\end{align}

\section*{Definitions}
\begin{equation}
    \text{diag}(\lambda_1, \lambda_2, \cdots, \lambda_N)=
    \begin{bmatrix}
    \lambda_1 &     0     & \dots  & 0 \\
       0      & \lambda_2 & \dots  & 0 \\
       \vdots &     \vdots     & \ddots & \vdots\\
    0         &     0     & \dots  &  \lambda_N
\end{bmatrix}.    
\end{equation}
